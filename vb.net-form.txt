    Private Function ValidateProjectChargecode(chargecode As String) As Boolean
        ' Check if the Project Chargecode meets the criteria
        chargecode = chargecode.ToLower()

        If chargecode.Length = 10 AndAlso (chargecode.Substring(0, 2) = "i-" OrElse chargecode.Substring(0, 2) = "e-" OrElse chargecode.Substring(0, 2) = "m-") Then
            ' Check if the chargeCode is different from the specific values
            If chargecode <> "i-12345678" AndAlso chargecode <> "e-12345678" AndAlso chargecode <> "m-12345678" Then
                ValidateProjectChargecode = True ' Project Charge Code meets the criteria
            Else
                ValidateProjectChargecode = False ' Project Charge Code is not allowed
            End If
        Else
            ValidateProjectChargecode = False ' Project Charge Code does not meet the criteria
        End If
    End Function
    Private Sub Button1_Click(sender As Object, e As EventArgs) Handles Button1.Click
        ' Capture a screenshot of the form
        g = Me.CreateGraphics()
        bmp = New Bitmap(Me.Size.Width, Me.Size.Height, g)
        Dim mg As Graphics = Graphics.FromImage(bmp)
        mg.CopyFromScreen(Me.Location.X, Me.Location.Y, 0, 0, Me.Size)

        ' Display captured screenshot in a dialog
        Dim printPreview As New PrintPreviewDialog()
        Dim printDoc As New PrintDocument()
        AddHandler printDoc.PrintPage, AddressOf Print_doc_page
        printPreview.Document = printDoc
        printPreview.ShowDialog()

        ' Gather user input
        Dim Project_Charge As String = TextProject_Charge.Text.Trim()
        ' Debugging output to check the value of Project_Charge
        Console.WriteLine("Project_Charge: " & Project_Charge)

        If ValidateProjectChargecode(Project_Charge) = True Then
            Dim Area1 As String = ComboBox4.Text
            Dim Project_Name As String = TextBox1.Text
            Dim Region As String = ComboBox5.Text
            Dim Current_state As String = ComboBox3.Text
            Dim deadline As String = DateTimePicker1.Text

            ' Display user input (you can replace this with your email sending logic)
            Dim userInput As String = $"Area is: {Area1}{vbCrLf}" &
                                      $"Project Name is: {Project_Name}{vbCrLf}" &
                                      $"Project charge code is: {Project_Charge}{vbCrLf}" &
                                      $"Region is: {Region}{vbCrLf}" &
                                      $"Technology state: {Current_state}{vbCrLf}" &
                                      $"Deadline is: {deadline}"

            MessageBox.Show(userInput)


            ' Send the email (uncomment this section when ready)
            Try
                SendEmail(Project_Charge, Area1, Project_Name, Region, Current_state, deadline)

            Catch ex As Exception
                MessageBox.Show($"Failed to send email: {ex.Message}")
            End Try

        Else
            MessageBox.Show("Invalid project charge code. Please make sure it's exactly 10 characters long and starts with 'I', 'E', or 'M'.")

    Private Sub SendEmail(projectCharge As String, area As String, projectName As String, region As String, techState As String, deadline As String)
        Dim smtpServer As New SmtpClient("smtp.gmail.com")
        Dim mail As New MailMessage()

        smtpServer.Port = 587
        smtpServer.Credentials = New Net.NetworkCredential("deepikayemeli@gmail.com", "Ss147896!")
        smtpServer.EnableSsl = True ' Use SSL/TLS

        smtpServer.Host = "smtp.gmail.com"
        mail = New MailMessage()
        mail.From = New MailAddress("deepikayemeli@gmail.com")
        mail.To.Add("deepikayemeli@gmail.com")
        mail.Subject = "Test Mail"
        mail.Body = $"Area: {area}{vbCrLf}" &
                    $"Project Name: {projectName}{vbCrLf}" &
                    $"Project Charge Code: {projectCharge}{vbCrLf}" &
                    $"Region: {region}{vbCrLf}" &
                    $"Technology State: {techState}{vbCrLf}" &
                    $"Deadline: {deadline}"

        smtpServer.Send(mail)
    End Sub
